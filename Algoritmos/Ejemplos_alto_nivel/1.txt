funcion flotante area_circulo(flotante radio) {
    flotante pi = 3.14159;
    retornar pi * radio * radio;
}

    flotante pi = 3.14159;
    flotante temp = 4.0 / 3.0;
    retornar temp * pi * radio * radio * radio;
}
