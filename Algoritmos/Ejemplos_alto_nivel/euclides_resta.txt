/*
 * Algoritmo de Euclides - Versión con Restas
 * Basado en la versión del profesor Peña
 * Calcula el Máximo Común Divisor (MCD) de dos números
 * usando el método de restas sucesivas
 */

funcion entero4 euclides_resta(entero4 a, entero4 b) {
    mientras (a != b) {
        si (a > b) {
            a = a - b;
        } si_no {
            b = b - a;
        }
    }
    retornar a;
}

funcion entero4 principal() {
    entero4 a = 1071;
    entero4 b = 462;
    entero4 resultado;
    
    imprimir("Algoritmo de Euclides (restas)");
    imprimir();
    imprimir("Calculando MCD de ");
    imprimir(a, b);
    imprimir();
    
    resultado = euclides_resta(a, b);
    
    imprimir("MCD = ");
    imprimir(resultado);
    imprimir();
    
    retornar resultado;
}
